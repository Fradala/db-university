SELECT `students`.`name`, `students`.`surname`
FROM `students`
JOIN `department` 
ON `students`.`id` = `department`.`id`
JOIN `courses` 
ON `departments`.`id` = `courses`.`id`
WHERE `courses`.`name` = 'Economia';

2
SELECT `courses`.`name`
FROM `courses`
JOIN `departments`
ON `courses`.`department_id` = `departments`.`department_id`
WHERE `departments`.`name`='Neuroscienze'
AND `courses`.`level` = 'Magistrale';

FROM `courses`
JOIN `teachers`
ON `courses`.`teachers_id`= `teachers`.`id`
WHERE `teachers`.`name` = 'Fulvio Amato';
